```python
answer = list(range(1, 10001))

def bruteforce(n):
    r = n%10
    q = int(n/10)
    dn = n + r
    while q>0:
        r = q%10
        q = int(q / 10)
        dn+=r
    try:
        answer.remove(dn)
    except:
        pass

for i in range(10001):
    bruteforce(i)

for elem in answer:
    print(elem)
```
